    \frametitle{Welcome!}
    Welcome to HCI
    \vskip25pt
    \begin{block}{Things we'll be covering:} 
        \begin{enumerate}
            \item Understand system \alert{usability}
            \item Design an \alert{efficient} UX
            \item Involving users in the design of systems
            \item \alert{Evaluate} interface usability and performance
            \item Use interface \alert{components} and \alert{design patterns}
            \item Use appropriate tools for handling both \alert{output} and \alert{input}
        \end{enumerate}
    \end{block}
}

\frame{
    \frametitle{Course contents}
    A preliminary list of topics (maybe not in this order):
    \begin{itemize}
        \item Usability
        \item Usability engineering \& task analysis
        \item Usability guidelines and heuristics
        \item Generating and prototyping designs
        \item Hardware interfaces \& I/O
        \item Software interfaces
        \item System-specific design (Android, iOS, wearable)
        \item Design patterns
        \item System evaluation 
    \end{itemize}
}
